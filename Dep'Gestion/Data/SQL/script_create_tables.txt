CREATE TABLE departement
(
	id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
	nom CHAR(250) NOT NULL,
	id_annee INT FOREIGN KEY REFERENCES annee(id)
)

CREATE TABLE annee
(
	id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
	nom CHAR(250) NOT NULL,
	id_partie_annee INT FOREIGN KEY REFERENCES partie_annee(id)
)

CREATE TABLE partie_annee
(
	id INT PRIMARY KEY NOT NULL,
	nom CHAR(250) NOT NULL,
	id_annee  INT  NOT NULL,
    FOREIGN KEY (id_annee)
    REFERENCES annee(id)
    ON DELETE CASCADE ON UPDATE CASCADE
)

CREATE TABLE enseignement
(
	id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
	nom CHAR(250) NOT NULL,
)

CREATE TABLE cours
(
	id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
	nb_groupes INT,
	id_type_cours INT NOT NULL
)

CREATE TABLE groupe
(
	id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
	nom CHAR(250) NOT NULL,
	id_enseignant INT NOT NULL,
	id_cours INT NOT NULL
)

CREATE TABLE type_cours
(
	id INT PRIMARY KEY NOT NULL,
	nom CHAR(250) NOT NULL,
	has_cours INT 
);


CREATE TABLE [dbo].[equivalent_td] (
                    [id]                      INT NOT NULL,
                    [id_categorie_enseignant] INT  NOT NULL,
                    [id_type_cours]           INT  NOT NULL,
                    [ratio_cours_td]          double NOT NULL,
                    PRIMARY KEY CLUSTERED ([id] ASC, [id_type_cours] ASC, [id_categorie_enseignant] ASC),
                    );

CREATE TABLE [dbo].[categorie_enseignant] (
    [id]                  INT PRIMARI KEY NOT NULL,
    [nom]                 VARCHAR (50) NOT NULL,
    [id_eqtd]             INT NOT NULL,
    [heures_a_travailler] double NOT NULL,
    PRIMARY KEY ([id] ASC), 
);

CREATE TABLE [dbo].[enseignant]
(
	id INT PRIMARY KEY NOT NULL,
	nom CHAR(250) NOT NULL,
	prenom CHAR(250) NOT NULL,
	nb_heures_assignees double NOT NULL,
	id_categorie_enseignant INT NOT NULL
);
